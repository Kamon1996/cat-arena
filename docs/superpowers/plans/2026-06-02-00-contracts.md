## 1. Final Prisma schema

Single source for `prisma/schema.prisma`. All ids are `cuid`. Relation fields added so the schema
compiles. Auth.js Prisma-adapter models (User, Account, Session, VerificationToken) included.

```prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider  = "postgresql"
  url       = env("DATABASE_URL")     // pooled (Neon PgBouncer) — runtime
  directUrl = env("DIRECT_URL")       // direct (non-pooled) — migrations
}

// ─────────────────────────── Domain: Cats ───────────────────────────

model Cat {
  id         String    @id @default(cuid())
  slug       String    @unique               // SEO URL: <name>-<nanoid>
  name       String                          // cat display name (UI + SEO)
  ownerId    String
  owner      User      @relation(fields: [ownerId], references: [id])
  status     CatStatus @default(PENDING)      // PENDING | ACTIVE | HIDDEN | BANNED
  // Glicko-2 — the CAT is rated, not an individual image
  rating     Float     @default(1500)
  rd         Float     @default(350)
  vol        Float     @default(0.06)
  score      Float     @default(800)          // = rating - 2*rd (denormalized for sorting)
  wins       Int       @default(0)
  losses     Int       @default(0)
  draws      Int       @default(0)            // reserved
  timesShown Int       @default(0)
  images     CatImage[]
  reports    Report[]
  orgs       CatOrg[]
  createdAt  DateTime  @default(now())
  approvedAt DateTime?

  @@index([status, score])                    // leaderboard
  @@index([status, rd])                        // "needs data" pool for pairing
  @@index([ownerId])                           // enforce ≤2 cats per user
}

model CatImage {
  id        String      @id @default(cuid())
  catId     String
  cat       Cat         @relation(fields: [catId], references: [id], onDelete: Cascade)
  r2Key     String                            // original; thumb/card derived by key convention
  width     Int
  height    Int
  position  Int         @default(0)           // carousel order
  status    ImageStatus @default(PENDING)     // moderation / NSFW at image level
  createdAt DateTime    @default(now())

  @@index([catId, position])
}

model Vote {
  id          String   @id @default(cuid())
  winnerCatId String                          // vote is BETWEEN cats
  loserCatId  String
  voterKey    String                          // anonId (cookie) or "user:<id>"
  createdAt   DateTime @default(now())

  @@index([voterKey, createdAt])              // seen-dedup / history
  @@index([winnerCatId])
  @@index([loserCatId])
}

model Report {                                // report against a cat
  id        String   @id @default(cuid())
  catId     String
  cat       Cat      @relation(fields: [catId], references: [id], onDelete: Cascade)
  reason    String?
  reporter  String                            // anonId / user
  createdAt DateTime @default(now())

  @@index([catId])
}

// ─────────────────────────── Domain: Organizations ───────────────────

model Organization {
  id          String   @id @default(cuid())
  name        String   @unique                // required, unique
  slug        String   @unique
  description String?                          // optional description
  logoR2Key   String?                          // optional logo (in R2)
  joinCode    String   @unique                // long generated invite code
  createdById String   @unique                // ≤1 org per user (unique creator)
  createdBy   User     @relation(fields: [createdById], references: [id])
  createdAt   DateTime @default(now())
  members     CatOrg[]
}

model CatOrg {                                 // membership + LOCAL Glicko-2 of a cat in an org
  catId      String
  orgId      String
  cat        Cat          @relation(fields: [catId], references: [id], onDelete: Cascade)
  org        Organization @relation(fields: [orgId], references: [id], onDelete: Cascade)
  rating     Float        @default(1500)
  rd         Float        @default(350)
  vol        Float        @default(0.06)
  score      Float        @default(800)        // org leaderboard sorts by score
  wins       Int          @default(0)
  losses     Int          @default(0)
  draws      Int          @default(0)
  timesShown Int          @default(0)
  joinedAt   DateTime     @default(now())

  @@id([catId, orgId])
  @@index([orgId, score])                      // org leaderboard
}

enum CatStatus   { PENDING ACTIVE HIDDEN BANNED }
enum ImageStatus { PENDING APPROVED REJECTED }

// ─────────────────────── Auth.js (Prisma adapter) ────────────────────

model User {
  id            String    @id @default(cuid())
  name          String?
  email         String?   @unique
  emailVerified DateTime?
  image         String?
  role          String    @default("user")    // "user" | "admin" (admin gate for /admin)
  accounts      Account[]
  sessions      Session[]
  cats          Cat[]                          // owned cats (Cat.owner relation)
  organization  Organization?                  // ≤1 org per user (Organization.createdBy relation)
  createdAt     DateTime  @default(now())
}

model Account {
  id                String  @id @default(cuid())
  userId            String
  type              String
  provider          String
  providerAccountId String
  refresh_token     String?
  access_token      String?
  expires_at        Int?
  token_type        String?
  scope             String?
  id_token          String?
  session_state     String?
  user              User    @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([provider, providerAccountId])
  @@index([userId])
}

model Session {
  id           String   @id @default(cuid())
  sessionToken String   @unique
  userId       String
  expires      DateTime
  user         User     @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@index([userId])
}

model VerificationToken {
  identifier String
  token      String   @unique
  expires    DateTime

  @@unique([identifier, token])
}
```

Notes for plan authors:
- A cat enters duels iff `Cat.status = ACTIVE` AND it has ≥1 `CatImage` with `status = APPROVED`.
- `Cat.score` and `CatOrg.score` are denormalized as `rating - 2*rd` and recomputed on every vote.
- `Vote.voterKey` holds either an anon cookie id or the string `"user:<id>"`.
- `Organization.createdById` is `@unique` — this is the DB-level enforcement of `MAX_ORGS_PER_USER = 1`.
- `MAX_ORGS_PER_CAT = 2` is enforced in API (count `CatOrg` rows for the cat before insert).

## 3. Module signatures (exact TypeScript)

### `src/rating/glicko2.ts` — pure, no DB

```ts
export type Rating = {
  rating: number; // μ — Glicko-2 rating (display scale, start 1500)
  rd: number;     // φ-equivalent rating deviation (display scale, start 350)
  vol: number;    // σ — volatility (start 0.06)
};

export type Opponent = {
  rating: number; // opponent rating (display scale)
  rd: number;     // opponent rating deviation (display scale)
  score: number;  // outcome vs this opponent: 1 = win, 0 = loss, 0.5 = draw
};

/**
 * Pure Glicko-2 update for one rating period.
 * Each vote is treated as a mini-period (single opponent in the array).
 * Applies a floor to rd so ratings stay responsive (see constants).
 * Returns the player's new Rating; does not mutate the input.
 */
export function updateGlicko2(player: Rating, opponents: Opponent[]): Rating;

/**
 * Conservative leaderboard score = rating - 2 * rd (lower bound of 95% CI).
 * Mirrors SCORE in src/lib/constants.ts.
 */
export function conservativeScore(r: Rating): number;
```

### `src/pairing/pick-pair.ts`

```ts
import type { Cat } from "@prisma/client";

export type PairScope = "global" | { orgId: string };

export type PickPairOptions = {
  scope: PairScope;
  seenCatIds: string[];   // cat ids to exclude (anon ring-buffer or Vote-derived dedup)
  voterKey: string;       // anonId or "user:<id>" — used for dedup/rate context
};

export type PickedPair = {
  a: Pick<Cat, "id" | "rating" | "rd" | "score" | "timesShown">;
  b: Pick<Cat, "id" | "rating" | "rd" | "score" | "timesShown">;
};

/**
 * Selects a duel pair.
 *  - scope "global": pool = ACTIVE cats with ≥1 APPROVED image.
 *  - scope { orgId }: pool restricted to that org's members (ACTIVE, ≥1 APPROVED).
 * A: weighted sample favoring high rd and low timesShown (weight ∝ rd · 1/√(timesShown+1)),
 *    drawn from a small candidate pool via the (status, rd) index — never full-table sort.
 * B: close to A by score (score BETWEEN A.score ± window via (status, score) index),
 *    excluding seenCatIds, with LIMIT + ε-randomness.
 * Returns null when no valid pair can be formed.
 */
export function pickPair(opts: PickPairOptions): Promise<PickedPair | null>;
```

### `src/lib/pair-token.ts`

```ts
export type PairTokenPayload = {
  a: string;     // cat id A
  b: string;     // cat id B
  nonce: string; // single-use nonce
  exp: number;   // unix epoch seconds — short expiry
  scope: string; // "global" or "org:<id>" — feed provenance
};

/** HMAC-sign the payload (secret = PAIR_TOKEN_SECRET). Returns the opaque token string. */
export function signPairToken(payload: PairTokenPayload): string;

/**
 * Verify signature and expiry. Returns the decoded payload if valid & unexpired,
 * otherwise null. Single-use enforcement (nonce consumption) is handled by the
 * caller via rate-limit / nonce store, not inside verify.
 */
export function verifyPairToken(token: string): PairTokenPayload | null;
```

### `src/lib/rate-limit.ts`

```ts
export type RateLimitResult = {
  ok: boolean;       // true if request is within the limit
  remaining: number; // tokens left in the current window
};

/**
 * Upstash Redis token-bucket check, keyed by voterKey or IP.
 * Consumes one token per call. Limits configured via constants.
 */
export function check(key: string): Promise<RateLimitResult>;
```

## 4. Shared named constants — `src/lib/constants.ts`

The single home for all shared magic values. No other file may inline these numbers.

```ts
// Entity limits (configurable defaults — may change later)
export const MAX_CATS_PER_USER = 2;
export const MAX_IMAGES_PER_CAT = 3;
export const MAX_ORGS_PER_USER = 1;
export const MAX_ORGS_PER_CAT = 2;

// Glicko-2 starting values (the rating entity is the Cat)
export const GLICKO_DEFAULT = {
  rating: 1500,
  rd: 350,
  vol: 0.06,
} as const;

// Conservative leaderboard score: lower bound of 95% CI.
// score = rating - 2 * rd  (denormalized into Cat.score / CatOrg.score)
export const SCORE = (rating: number, rd: number): number => rating - 2 * rd;

// Image processing sizes (max edge, px) — sharp → WebP
export const IMAGE_SIZE = {
  THUMB: 200,
  CARD: 800,
} as const;
```

Plan authors: `Cat.rating/rd/vol/score` Prisma defaults (1500/350/0.06/800) MUST equal `GLICKO_DEFAULT`
applied through `SCORE` (1500 − 2·350 = 800). Same for `CatOrg`.

## 6. API route contracts

All handlers: validate body/query with **Zod at the boundary**, return typed JSON, set appropriate
HTTP status. Error response shape is uniform: `{ error: string }` with the listed status code.

### `GET /api/pair`
- **Method:** GET
- **Query (Zod):** `{ scope?: "global" | string }` — omitted/`"global"` → global pool; otherwise an org id (resolves to `{ orgId }`).
- **Behavior:** reads anon `seenCatIds` from signed cookie (or `Vote` dedup for logged-in), calls `pickPair`, signs a `PairTokenPayload` via `signPairToken` (scope = `"global"` or `"org:<id>"`).
- **Success 200:**
  ```ts
  {
    token: string;            // signed pair token
    a: { id: string; name: string; slug: string; images: { url: string; width: number; height: number; position: number }[] };
    b: { id: string; name: string; slug: string; images: { url: string; width: number; height: number; position: number }[] };
  }
  ```
- **Errors:** `400` invalid query · `404` no eligible pair available (empty/insufficient pool) · `500` internal.

### `POST /api/vote`
- **Method:** POST
- **Body (Zod):**
  ```ts
  { token: string; winnerCatId: string; loserCatId: string }
  ```
- **Behavior:** `verifyPairToken(token)` → must be valid, unexpired, and `{winnerCatId, loserCatId}` must equal `{a,b}` (in either order); enforce single-use (nonce) and `rate-limit.check(voterKey/IP)`; apply Glicko-2 to both cats (global) and to each shared org in `orgs(W) ∩ orgs(L)` when `scope = org:<id>`; update wins/losses/timesShown and recompute `score`; insert `Vote`. Idempotent via single-use token.
- **Success 200:**
  ```ts
  {
    ok: true;
    winner: { id: string; rating: number; rd: number; score: number };
    loser:  { id: string; rating: number; rd: number; score: number };
  }
  ```
- **Errors:** `400` invalid body · `401`/`403` invalid/expired/mismatched or replayed token · `429` rate-limited · `500` internal.

### `POST /api/upload/sign`
- **Method:** POST · **auth required**
- **Body (Zod):**
  ```ts
  { contentType: "image/jpeg" | "image/png" | "image/webp"; size: number /* bytes, ≤ 10MB */ }
  ```
- **Behavior:** validates type & size; generates `r2Key`; returns presigned PUT URL.
- **Success 200:**
  ```ts
  { uploadUrl: string; r2Key: string }
  ```
- **Errors:** `400` bad type/size · `401` not authenticated · `500` internal.

### `POST /api/cats`
- **Method:** POST · **auth required**
- **Body (Zod):**
  ```ts
  {
    name: string;                         // 1..N chars
    images: { r2Key: string }[];          // 1..MAX_IMAGES_PER_CAT, in carousel order
    joinCode?: string;                    // optional org invite code
  }
  ```
- **Behavior:** enforce `MAX_CATS_PER_USER` for the owner and `images.length ≤ MAX_IMAGES_PER_CAT`; for each image fetch original from R2, run `sharp` (strip EXIF → WebP, thumb 200 / card 800), write derivatives, optional NSFW screen → `ImageStatus`; create `Cat` (status `PENDING`, slug `<name>-<nanoid>`) + `CatImage[]` with `position`; if a clean APPROVED image exists, promote `Cat.status` to `ACTIVE`; if `joinCode` present, attempt org join (≤ `MAX_ORGS_PER_CAT`).
- **Success 201:**
  ```ts
  { id: string; slug: string; status: CatStatus }
  ```
- **Errors:** `400` invalid body / too many images · `401` not authenticated · `409` cat limit reached (`MAX_CATS_PER_USER`) · `422` invalid `joinCode` · `500` internal.

### `POST /api/orgs`
- **Method:** POST · **auth required**
- **Body (Zod):**
  ```ts
  {
    name: string;            // required, unique
    description?: string;    // optional
    logoR2Key?: string;      // optional logo already uploaded to R2
  }
  ```
- **Behavior:** enforce `MAX_ORGS_PER_USER = 1` (DB unique on `createdById`); generate unique `joinCode` and `slug`; create `Organization`.
- **Success 201:**
  ```ts
  { id: string; slug: string; joinCode: string }
  ```
- **Errors:** `400` invalid body · `401` not authenticated · `409` user already owns an org (`MAX_ORGS_PER_USER`) or name not unique · `500` internal.

### `POST /api/cats/[id]/orgs`  (join by code)
- **Method:** POST · **auth required (cat owner)**
- **Path param:** `id` = cat id
- **Body (Zod):**
  ```ts
  { joinCode: string }
  ```
- **Behavior:** verify caller owns cat `id`; resolve `joinCode` → `Organization`; if cat has `< MAX_ORGS_PER_CAT` memberships, create `CatOrg` (local Glicko defaults). Valid code alone suffices — no org-owner approval.
- **Success 200:**
  ```ts
  { ok: true; orgId: string; orgSlug: string }
  ```
- **Errors:** `400` invalid body · `401` not authenticated · `403` not the cat owner · `404` cat not found · `409` cat already at `MAX_ORGS_PER_CAT` or already a member · `422` invalid `joinCode` · `500` internal.

### `DELETE /api/cats/[id]/orgs`  (leave)
- **Method:** DELETE · **auth required (cat owner)**
- **Path param:** `id` = cat id
- **Body (Zod):**
  ```ts
  { orgId: string }
  ```
- **Behavior:** verify caller owns cat `id`; delete the `CatOrg` row for `(catId=id, orgId)`.
- **Success 200:**
  ```ts
  { ok: true }
  ```
- **Errors:** `400` invalid body · `401` not authenticated · `403` not the cat owner · `404` membership not found · `500` internal.

### `POST /api/report`
- **Method:** POST  (voting is anonymous; reporter may be anon or user)
- **Body (Zod):**
  ```ts
  { catId: string; reason?: string }
  ```
- **Behavior:** resolve `reporter` (anonId or `"user:<id>"`); insert `Report`; if cat crosses the report threshold, auto-set `Cat.status = HIDDEN`.
- **Success 201:**
  ```ts
  { ok: true }
  ```
- **Errors:** `400` invalid body · `404` cat not found · `429` rate-limited · `500` internal.
